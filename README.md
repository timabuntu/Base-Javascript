## <a name="parte3">3 - Variáveis e dados</a>


```javascript
<!DOCTYPE html>
<html lang="en">
<head>
    <title>Curso JavaScript - Rocketseat</title>
</head>
<body>
    
    <script>
    let nome = "José";
    let idade = 34;
    let peso = 90.3;
    let humano = true;

    let alunos = ['jose','marcos','joão'];
    let aluno = {
        nome: alunos[0],
        idade: 20,
        peso: 80.8,
        humano: true,
    }
    console.log(alunos);
    console.log(alunos[1]);

    console.log(aluno.nome);
    console.log(aluno.peso);
    </script>
</body>
</html>
```

## <a name="parte5">5 - Funções</a>

```javascript
<!DOCTYPE html>
<html lang="en">

<head>
  <title>Curso JavaScript - Rocketseat</title>
</head>

<body>

  <script>
    function soma(numero1, numero2) {
      let resultado = numero1 + numero2;
      return resultado;
    }
    var resultado = soma(10, 200);
    console.log(resultado);
  </script>
</body>

</html>
```
